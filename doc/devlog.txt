# Application Setup
###################
rails lifeline
cd lifeline

# mac development environment install
sudo port install sqlite3
sudo gem install sqlite3-ruby

# linux development environment install
# to get gem to work, you may need to install it manually (i.e. download tar, run "sudo ruby setup.rb" in the gem directory
sudo gem update --system
sudo gem install sqlite3-ruby

# windows development environment
# to get sqlite3 to work
# download sqlite3.exe and sqlite3.dll (from sqlite.org) --> copy to windows\system32 dir
# download precompiled win32 gem for sqlite3-ruby from rubyforge
gem install (full win32 gem)

rake db:migrate

ruby script/generate model user
ruby script/generate model group
ruby script/generate model msg
ruby script/generate model org

# Gems and Plugins
###################
# if you have any missing plugins use the following
sudo rake gems:install

# to include gems with the system
# add gem to environment.rb
  config.gem 'mislav-will_paginate', :version => '~> 2.3.2', :lib => 'will_paginate', :source => 'http://gems.github.com'
  config.gem 'clickatell', :source=>'git://github.com/lukeredpath/clickatell.git'

# use rake task to unpack them
rake gems:unpack GEM=clickatell

# how to MANUALLY install the clickatell
mkdir clickatell
cd clickatell
git init
git pull git://github.com/lukeredpath/clickatell.git
cd ..
sudo gem install clickatell


sudo gem sources -a http://gems.github.com
