require "clickatell"

API_ID    = 'API_ID'
username,password = ['USERACCT','PASSWORD']

api = Clickatell::API.authenticate( API_ID, username, password )

phone_no = '16175551111'
phone,text = [phone_no,'hello from clickatell']
message_id = api.send_message( phone, text )
status_code = api.message_status( message_id )

bal = api.account_balance
